 16 /**
 17  * @class 해당 콤포넌트의 form으로 부터의 경로를 구하는 함수
 18  * @param {Object} obj - 콤포넌트
 19  * @return {String} 해당 콤포넌트의 form으로 부터의 경로
 20  */
 21 this.gfnGetCompId = function (obj)
 22 {
 23 	var sCompId = obj.name;
 24 	var objParent = obj.parent;
 25 	
 26 	while (true)
 27 	{
 28 		//trace("" + objParent + " / " + objParent.name);
 29 		if (objParent instanceof nexacro.ChildFrame )
 30 		{
 31 			break;
 32 		}
 33 		else {
 34 			sCompId = objParent.name + "." + sCompId;
 35 		}
 36 		objParent = objParent.parent;		
 37 	}
 38 	return sCompId;
 39 }

 41 /**
 42  * @class 월력용 Calendar에 월력 Popup Div을 자동생성하고 해당 Popup Div을 호출
 43  * @param {Object} obj - 월력용 Calendar
 44  * @return N/A
 45  */
 46 this.gfnCalMMOndropdown = function (obj)
 47 {
 48 	var pdvName = this.gfnGetCompId(obj);
 49 	
 50 	// Creating pdv
 51 	if (this.gfnIsNull(this.components[pdvName])) {		
 52 		var objCalPopupDiv = new PopupDiv();
 53 		objCalPopupDiv.init(pdvName, obj.getOffsetLeft(), obj.getOffsetBottom(), 180, 200, null, null);		
 54 		this.addChild(pdvName, objCalPopupDiv);
 55 		objCalPopupDiv.show();
 56 		objCalPopupDiv.set_url("cmm::cmmCalMMPdv.xfdl");
 57 		objCalPopupDiv.calObj = obj;
 58 	}
 59 	else {
 60 		var objCalPopupDiv = this.components[pdvName];
 61 		
 62 		// Calendar에서 수정한 년도를 Popup Div에 반영
 63 		var sDate = obj.value;
 64 		objCalPopupDiv.form.staYYYY.set_text(sDate.substr(0,4));
 65 	}
 66 	//trace("pdvName : " + pdvName);
 67 
 68 	var nLeft = 0; 	
 69 	// Compoent가 오른쪽에 있을 경우 Component와 우측 정렬하여 팝업Div 표시
 70 	if (this.getOffsetWidth() < (obj.getOffsetRight() + 180) ) {
 71 		nLeft = obj.getOffsetWidth() - 180;
 72 	}
 73 	else {
 74 		nLeft = 0; 
 75 	}
 76 	
 77 	var nTop = 0;	
 78 	// Compoent가 아래쪽에 있을 경우 Component 위로 팝업Div 표시
 79 	if (this.getOffsetHeight() < (obj.getOffsetBottom() + 200)) {
 80 		nTop = -200;
 81 	}
 82 	else {
 83 		nTop = obj.getOffsetHeight(); 
 84 	}
 85 	objCalPopupDiv.trackPopupByComponent(obj, nLeft, nTop);
 86 	
 87 	return false;
 88 };

 90 /**
 91  * @class Div를 기준으로 Header/Body 의 쌍이 접고 펼치는 처리를 하는 공통함수
 92  * @param {Object} objList - 아코디언처리할 대상 Object의 목록으로 JSON Object의 형식으로 Header : Body의 형식으로 작성한다.
 93  * @param {Boolean or Integer} openMode - 아코디언 초기 설정값으로 true: 전체 보여주기, false: 전체 숨기기, index: 해당항목만 펼쳐준다.
 94  * @return N/A
 95  */
 96 this.gfnSetAccordion = function(objList, openMode)
 97 {
 98     var targetlist = new Array();
 99 
100     for(var objName in objList)
101     {
102 		var targetobj = eval("this."+objName);
103 
104         // 대상아코디언이 있으면
105         if( !this.gfnIsNull(objList[objName]) )
106         {
107             // Div Object일경우
108             if( objList[objName].valueOf() == "[object Div]" )
109             {
110                 targetobj.accordionbody = objList[objName];
111             }
112             // String으로 넘어온경우
113             else
114             {
115                 targetobj.accordionbody = eval("this."+objList[objName]);
116             }
117 
118             targetobj.accordionbody.orgheight = targetobj.accordionbody.height;
119         }
120 
121         targetlist.push(targetobj);
122     }
123 
124     // 각각의 Object의 목록에 대한 아코디언설정정보를 설정한다.
125     var firstbtnobj;
126     for(var i=0;i<targetlist.length;i++)
127     {
128         // 어코디언 body 부분이 없는것은 무조건 보여준다.
129         if( this.gfnIsNull(targetlist[i].accordionbody) )
130         {
131             targetlist[i].accordionopenstatus = true;
132         }
133 
134         var headobjects = targetlist[i].form.components;
135         for(var j=0;j<headobjects.length;j++)
136         {
137             // 아코디언 Close/Open 처리 버튼 검색 및 버튼에 필요한 아코디언 및 상태를 설정한다.
138             if( headobjects[j].valueOf() == "[object Button]" && headobjects[j].cssclass.indexOf("btn_WF_Acc") != -1 )
139             {
140                 // 아코디언 이벤트 처리핸들
141                 headobjects[j].addEventHandler("onclick", this._gfnHandleAccordionButtonClick, this);
142                 headobjects[j].accordionlist = targetlist;
143                 headobjects[j].accordionhead = targetlist[i];
144                 //headobjects[j].setFocus(false);
145 
146                 if( this.gfnIsNull(firstbtnobj) )
147                 {
148                     firstbtnobj = headobjects[j];
149                 }
150 
151                 if( this.gfnIsBoolean(openMode))
152                 {
153                     headobjects[j].accordionopenstatus = openMode;
154 
155                     if( openMode == true )
156                     {
157                         headobjects[j].set_cssclass("btn_WF_AccClose");
158 						headobjects[j].set_text("CLOSE");
159                     }
160                     else
161                     {
162                         headobjects[j].set_cssclass("btn_WF_AccOpen");
163 						headobjects[j].set_text("OPEN");
164                     }
165 
166                     targetlist[i].accordionopenstatus = openMode;
167                 }
168                 else
169                 {
170                     if( openMode == i )
171                     {
172                         headobjects[j].accordionopenstatus = true;
173                         headobjects[j].set_cssclass("btn_WF_AccClose");
174 						headobjects[j].set_text("CLOSE");
175                         targetlist[i].accordionopenstatus = true;
176                     }
177                     else
178                     {
179                         headobjects[j].accordionopenstatus = false;
180                         headobjects[j].set_cssclass("btn_WF_AccOpen");
181 						headobjects[j].set_text("OPEN");
182                         targetlist[i].accordionopenstatus = false;
183                     }
184                 }                
185                 break;
186             }
187         }
188     }
189 
190     // 현재의 상태에 맞게 리드로잉처리한다.
191     this._gfnRedrawAccordion(firstbtnobj);
192 };

224 /**
225  * @class 아코디언의 상태값을 이용해서 리드로잉 처리한다.
226  * @param {Button}         아코디언 해더에 존재하는 버튼
227  * @return N/A
228  */
229 this._gfnRedrawAccordion = function(obj)
230 {
231 	//trace("===== _gfnRedrawAccordion : " + obj.name);
232     var heightoffset  = 0;
233     var accordionlist = obj.accordionlist;
234 
235     var toppos = accordionlist[0].top;
236     for(var i=0;i<accordionlist.length;i++)
237     {
238 		//trace("===== _gfnRedrawAccordion accordionlist[i].name : " + accordionlist[i].name);
239 		
240         accordionlist[i].set_top(toppos);
241 
242         // 보여주기 처리
243         if( accordionlist[i].accordionopenstatus == true )
244         {
245             // 바디가 존재할때
246             if( !this.gfnIsNull(accordionlist[i].accordionbody) )
247             {
248                 accordionlist[i].accordionbody.set_top(parseInt(toppos)+parseInt(accordionlist[i].height));
249                 accordionlist[i].accordionbody.set_height(accordionlist[i].accordionbody.orgheight);
250                 accordionlist[i].accordionbody.set_visible(true);
251 				accordionlist[i].parent.resetScroll();
252                 toppos = parseInt(toppos)+parseInt(accordionlist[i].height)+parseInt(accordionlist[i].accordionbody.height)+parseInt(heightoffset);
253             }
254             else
255             {
256                 toppos = parseInt(toppos)+parseInt(accordionlist[i].height)+parseInt(heightoffset);
257             }
258         }
259         // 숨김처리
260         else
261         {
262             if( !this.gfnIsNull(accordionlist[i].accordionbody) )
263             {
264                 accordionlist[i].accordionbody.set_top(parseInt(toppos)+parseInt(accordionlist[i].height));
265                 accordionlist[i].accordionbody.set_height(0);
266                 accordionlist[i].accordionbody.set_visible(false);
267                 accordionlist[i].parent.resetScroll();
268                 toppos = parseInt(toppos)+parseInt(accordionlist[i].height)+parseInt(heightoffset);
269             }
270             else
271             {
272                 toppos = parseInt(toppos)+parseInt(accordionlist[i].height)+parseInt(heightoffset);
273             }
274         }
275     }
276 };

461 /**
462  * @class  DLL의 full 경로를 리턴
463  * @param  {String} 컴포넌트 DLL 파일명
464  * @return {String} 해당 DLL의 실제 경로
465  */
466 this.gfnGetDLLPath = function(sFileName)
467 {
468 	var strpath = "";
469 	var xadl = nexacro.getProjectPath();
470 	
471 	// studio로 실행
472     if (xadl.indexOf("file://") != -1) {        
473 		xadl = xadl.replace("file://", "");
474         strpath  = nexacro.replaceAll(xadl.substring(0, xadl.lastIndexOf("/")), "/", "\\")+"\\dll\\"+sFileName;
475     }
476 	// 웹 접속
477     else {
478 		strpath = system.convertRealPath("%USERAPP%\\Component\\" + sFileName);
479 	}
480 	
481 	//trace("==================== gfnGetDLLPath strpath : " + strpath);
482     return strpath;
483 };

486  * @class 컴포넌트 DLL 을 로딩하고 그 결과 Object를 리턴
487  * @param {String}  컴포넌트 DLL 파일명
488  * @return {Object} DLL 컴포넌트 Object
489  */ 
490 this.gfnOpenDLL = function(sFileName)
491 {
492 	var objApp = this.gfnGetApplication();
493 	
494 	if (this.gfnIsNull(objApp[sFileName])) {
495 		objApp[sFileName] = {};
496 	}
497 	else {
498 		return objApp[sFileName];
499 	}
500 
501 	// 로컬/웹 접근에 따른 파일 full 경로 가져오기
502 	var strpath = this.gfnGetDLLPath(sFileName);
503 
504 	// dll load
505 	objApp[sFileName] = nexacro._addExtensionModule(strpath);
506 
507 	return objApp[sFileName];
508 };
