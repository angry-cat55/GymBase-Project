 16 var pForm = nexacro.Form.prototype;
 17 
 18 /**
 19  * @class 팝업오픈
 20  * @param {String} sPopupId	- 팝업ID
 21  * @param {String} sUrl	 - 팝업URL
 22  * @param {String} [oArg] - 전달값
 23  * @param {String} [sPopupCallback] - 팝업콜백
 24  * @param {Object} [oOption] - 팝업옵션 <br>
 25  *	oOption.top : 상단 좌표 <br>
 26  *	oOption.left : 좌측 좌표 <br>
 27  *	oOption.width : 넓이 <br>
 28  *	oOption.height : 높이 <br>
 29  *	oOption.popuptype : 팝업종류(modal:showModal, modeless:application.open, modalsync:showModalSync, modalwindow:showModalWindow) <br>
 30  *	oOption.layered : 투명 윈도우 <br>
 31  *	oOption.opacity : 투명도 <br>
 32  *	oOption.autosize : autosize <br>
 33  * @return N/A
 34  * @example
 35  * this.gfnOpenPopup(this);
 36  */
 37 this.gfnOpenPopup = function ( sPopupId, sUrl, oArg, sPopupCallback, oOption)
 38 {
 39     var objApp = this.gfnGetApplication();
 40 	var nLeft = -1;
 41 	var nTop = -1;
 42 	var nWidth = -1;
 43 	var nHeight = -1;
 44 	var bShowTitle = true;	
 45 	var bShowStatus = false;	
 46 	var sPopupType = "modal";
 47 	var bLayered = false;
 48 	var nOpacity = 100;
 49 	var bAutoSize = false;
 50 	var bResizable = false;
 51 	//var sPopupCallback = (this.gfn_isNull(sPopupCallback)) ? "fn_popupAfter" : sPopupCallback;
 52 	var sTitleText = "";
 53 
 54 	for (var key in oOption) {
 55        if (oOption.hasOwnProperty(key)) {
 56             switch (key) 
 57 			{
 58 				case "top":				
 59 					nTop = parseInt(oOption[key]);
 60 					break;
 61 				case "left":
 62 					nLeft = parseInt(oOption[key]);
 63 					break;
 64 				case "width":
 65 					nWidth = parseInt(oOption[key]);
 66 					break;
 67 				case "height":
 68 					nHeight = parseInt(oOption[key]);
 69 					break;
 70 				case "popuptype":
 71 					sPopupType = oOption[key];
 72 					break;
 73 				case "layered":
 74 					bLayered = oOption[key];
 75 					break;
 76 				case "opacity":
 77 					nOpacity =oOption[key];
 78 					break;
 79 				case "autosize":
 80 					bAutoSize = oOption[key];
 81 					break;
 82 				case "titlebar":
 83 					if (""+oOption[key] == "false")	bShowTitle = false;		
 84 					break;
 85 				case "title":					
 86 					sTitleText = oOption[key];	
 87 					break;					
 88 			}	
 89         }
 90     }
 91 
 92 	var sOpenalign = "";
 93 	if(nLeft == -1 && nTop == -1) 
 94 	{		
 95 		sOpenalign = "center middle";
 96 		if (system.navigatorname == "nexacro") {
 97 			var curX = objApp.mainframe.left;
 98 			var curY = objApp.mainframe.top;
 99 		}else{
100 			var curX = window.screenLeft;
101 			var curY = window.screenTop;
102 		}
103 		
104         nLeft   =  curX + (objApp.mainframe.width / 2) - Math.round(nWidth / 2);
105 	    nTop    = curY + (objApp.mainframe.height / 2) - Math.round(nHeight / 2) ;		
106 		
107 	}else{
108 		nLeft   =  this.getOffsetLeft() + nLeft;
109 		nTop   =  this.getOffsetTop() + nTop;
110 	}
111 		
112 	if(nWidth == -1 || nHeight == -1)
113 	{
114 	    bAutoSize = true;
115 	}
116 	
117 	var objParentFrame = this.getOwnerFrame();
118 
119     if(sPopupType == "modeless")
120     {
121         var sOpenStyle= "showtitlebar=true showstatusbar=false showontaskbar=true showcascadetitletext=false resizable=true autosize="+bAutoSize+" titletext="+sTitleText;
122 		var arrPopFrame = nexacro.getPopupFrames();
123 
124 		if (arrPopFrame[sPopupId]) {	
125 			if (system.navigatorname == "nexacro") {
126 				arrPopFrame[sPopupId].setFocus();
127 			} else {	
128 				arrPopFrame[sPopupId]._getWindowHandle().focus();
129 			}
130 		}
131 		else {
132 			nexacro.open(sPopupId,sUrl,objParentFrame,oArg,sOpenStyle,nLeft, nTop, nWidth, nHeight, this);
133 		}
134     }
135 	else if(sPopupType == "modalsync")
136     {
137 		newChild = new nexacro.ChildFrame;
138 		newChild.init(sPopupId, nLeft, nTop, nWidth, nHeight, null, null, sUrl);
139 		
140 		newChild.set_dragmovetype("all");
141 		newChild.set_showcascadetitletext(false);
142 		newChild.set_showtitlebar(bShowTitle);    //titlebar는 안보임
143 		newChild.set_autosize(bAutoSize);	
144 		newChild.set_resizable(bResizable);    //resizable 안됨
145 		if(!this.gfnIsNull(sTitleText)) newChild.set_titletext(sTitleText);
146 		newChild.set_showstatusbar(bShowStatus);    //statusbar는 안보임
147 		newChild.set_openalign(sOpenalign);
148 		newChild.set_layered(bLayered);
149 		newChild.set_overlaycolor("RGBA(0, 0, 0, 0.2)");
150 		system.showModalSync(newChild, objParentFrame, oArg);	
151 	}
152 	else if(sPopupType == "modalwindow")
153     {
154 		newChild = new nexacro.ChildFrame;
155 		newChild.init(sPopupId, nLeft, nTop, nWidth, nHeight, null, null, sUrl);
156 		
157 		newChild.set_dragmovetype("all");
158 		newChild.set_showcascadetitletext(false);
159 		newChild.set_showtitlebar(bShowTitle);    //titlebar는 안보임
160 		newChild.set_autosize(bAutoSize);	
161 		newChild.set_resizable(bResizable);    //resizable 안됨
162 		if(!this.gfnIsNull(sTitleText)) newChild.set_titletext(sTitleText);
163 		newChild.set_showstatusbar(bShowStatus);    //statusbar는 안보임
164 		newChild.set_openalign(sOpenalign);
165 		newChild.set_layered(bLayered);
166 		newChild.set_overlaycolor("RGBA(0, 0, 0, 0.2)");
167 		var rtn = system.showModalWindow(newChild, sPopupId, objParentFrame, oArg);		
168         return rtn;
169 //         var objCF = new ChildFrame();
170 // 
171 // 		objCF .init("CF_modal", 0, 0, 500, 500);
172 // 		objCF .set_formurl("cmm::cmmPopup.xfdl");
173 // 		var ret = system.showModalWindow(objCF , "CF_modal", this.getOwnerFrame(), {a:'aaa', b:'bbb'}, this);
174 
175 	}	
176     else
177     {
178 		newChild = new nexacro.ChildFrame;
179 		newChild.init(sPopupId, nLeft, nTop, nWidth, nHeight, null, null, sUrl);
180 		
181 		newChild.set_dragmovetype("all");
182 		newChild.set_showcascadetitletext(false);
183 		newChild.set_showtitlebar(bShowTitle);    //titlebar는 안보임
184 		newChild.set_autosize(bAutoSize);	
185 		newChild.set_resizable(bResizable);    //resizable 안됨
186 		if(!this.gfnIsNull(sTitleText)) newChild.set_titletext(sTitleText);
187 		newChild.set_showstatusbar(bShowStatus);    //statusbar는 안보임
188 		newChild.set_openalign(sOpenalign);
189 		newChild.set_layered(bLayered);
190 		newChild.set_overlaycolor("RGBA(0, 0, 0, 0.2)");
191 		newChild.showModal(objParentFrame, oArg, this, this[sPopupCallback]);
192 		//newChild.titlebar.closebutton.set_visible(false);	//close버튼 visible false 처리
193 		//newChild.style.set_border("2 solid #24322b");
194 		//newChild.style.set_bordertype("round 10 10");
195 		//newChild.style.set_background("white");    
196 		//newChild.style.set_opacity(nOpacity);
197 
198     }
199 };
