 36 /**
 37  * @class Dataset의 Column별로 설정된 정합성체크 Rule을 제거한다.
 38  * @param  {Dataset} 데이터셋 Object

 42 this.gfnRemoveValidation = function(obj, sColumID)
 43 {
 44 	var newvalidObj = new Array();
 45 
 46 	if (!this.gfnIsNull(obj.validationRuleSet))
 47 	{
 48 		validObj = JSON.parse(obj.validationRuleSet);
 49 
 50 		for(var index in validObj)
 51 		{
 52 			var validationRule = validObj[index];
 53 
 54 			// 제거 대상이 아닐때만 정합성체크 대상에 넣는다.
 55 			if (validationRule.name != sColumID)
 56 			{
 57 				newvalidObj.push(validationRule);
 58 			}
 59 		}
 60 	}
 61 				
 62 	obj.validationRuleSet = JSON.stringify(newvalidObj);            
 63 };

 65 /**
 66  * @class Dataset의 Column별로 정합성체크 Rule을 등록한다.
 67  * @param {Dataset} obj - 데이터셋 Object
 68  * @param {String} sColumID - 체크대상 컬럼명
 69  * @param {String} sColumName - 컬럼명의 오류메세지 처리 제목
 70  * @param {String} sValidRule - 정합성체크 Rule(required, length, min, max, code 등의 체크 형식)
 71  * @return N/A
 72  */
 73 this.gfnSetValidation = function(obj, sColumID, sColumName, sValidRule)
 74 {
 75 	var validObj = new Array();
 76 	
 77 	// title이 2개가 올 경우 구분자를 변경
 78 	// checkrule == "equalto" || checkrule == "fromto" || checkrule == "comparebig" || checkrule == "comparesmall"
 79 	if (sColumName.indexOf(",") != -1) sColumName = nexacro.replaceAll(sColumName, ",", "^_^");
 80     
 81 	// Dataset에 RuleSet이 존재하면
 82 	if (!this.gfnIsNull(obj.validationRuleSet))
 83 	{
 84 		validObj = JSON.parse(obj.validationRuleSet);
 85 		for(var index in validObj)
 86 		{
 87 			// 컬럼에 Rule이 존재하면 변경 처리
 88 			if( validObj[index].name == sColumID )
 89 			{            
 90 				validObj[index].title = sColumName;
 91 				validObj[index].value = sValidRule;
 92 				if (""+sValidRule.indexOf("required") > -1) {
 93 					validObj[index].notnull = true;
 94 				}
 95 				else {
 96 					validObj[index].notnull = false;
 97 				}
 98 				obj.validationRuleSet = JSON.stringify(validObj);
 99 				return true;
100 			}
101 		}
102 	}
103 	
104 	// Rule 지정
105 	var validationRule = 
106 	{
107 		 name  : sColumID		// Column
108 		,title : sColumName		// Title
109 		,value : sValidRule		// 정합성 Rule
110 	};
111 	// Rule에 필수여부 추가
112 	if (""+sValidRule.indexOf("required") > -1) {
113 		validationRule.notnull = true;
114 	}
115 	else {
116 		validationRule.notnull = false;
117 	}				
118 	validObj.push(validationRule);
119 
120 	obj.validationRuleSet = JSON.stringify(validObj);
121 };

124 /**
125  * @class Dataset에 등록된 데이터 정합성체크 RuleSet에의해 정합성을 체크를 하고 이상여부를 리턴한다.
126  * @param  {Dataset||Grid,String} Check 대상 Object(Dataset, Grid)
127                                   Check 모드 A: 모든 row에 대해서 체크, U: 변경된 row에 대해서만 체크(default)
128  * @return {Boolean} 정상 true / 오류 false
129  */
130 this.gfnValidation = function()
131 {
132 	var updatecount = 0;
133 	var targetlist  = new Array();
134 	var checkmode   = "U";
135 
136 	// Check 대상과 Check Mode(A/U)를 지정
137 	for(var m=0;m<arguments.length;m++)
138 	{
139 		// Check 모드(A-전체 row, U-변경된 row) 설정
140 		if (this.gfnTypeOf(arguments[m]) == "string")
141 		{
142 			if( arguments[m] == "A" || arguments[m] == "U" )
143 			{
144 				checkmode = arguments[m];
145 			}
146 		}
147 		// Check 대상 설정 : 데이타셋 or 그리드
148 		else
149 		{
150 			targetlist.push(arguments[m]);
151 		}
152 	}
153 
154 	// arguments로 넘어온 체크 대상 만큼 정합성 체크
155 	for (var m=0;m<targetlist.length;m++)
156 	{
157 		var dataset;
158 		var objtype   = "";
159 		var checklist = new Array();
160 
161 		// 데이터셋
162 		if( targetlist[m] instanceof Dataset ) {
163 			dataset = targetlist[m];
164 			objtype = "dataset";
165 		}
166 		// 그리드
167 		else if( targetlist[m] instanceof Grid ) {
168 			dataset = targetlist[m].getBindDataset();
169 			objtype = "grid";
170 		}
171 		// 처리 제외
172 		else {
173 			continue;
174 		}
175 
176 		// 데이터셋의 RuleSet
177 		var validlist = new Array();
178 		if( !this.gfnIsNull(dataset.validationRuleSet) )
179 		{
180 			validlist = JSON.parse(dataset.validationRuleSet);
181 		}
182 		else {
183 			continue;
184 		}
185 
186 		// 삭제된 row 정합성 체크 건수 증가
187 		updatecount += parseInt(dataset.getDeletedRowCount());
188 
189 		// 데이터셋의 row 만큼 정합성 체크
190 		for(var i=0;i<dataset.getRowCount();i++)
191 		{
192 			var rowtype = dataset.getRowType(i);
193 
194 			// 삭제된 row
195 			if (rowtype == 8) {
196 				++updatecount;
197 				continue;
198 			}
199 
200 			// checkmode에 따른 체크 대상 구분
201 			if (checkmode == "U") {
202 				// 변경 안된 Row는 정합성 체크 SKIP
203 				if (rowtype == 1) {
204 					continue;
205 				}
206 
207 				// 추가, 수정은 정합성 체크
208 				if (rowtype == 2 || rowtype == 4) {
209 					++updatecount;
210 				}
211 			}
212 			// 전체 row
213 			else {
214 				++updatecount;
215 			}
216 
217 			// Rule 만큼 처리
218 			for(var j=0;j<validlist.length;j++)
219 			{
220 				var columid   = validlist[j].name;
221 				var columname = new String(validlist[j].title);
222 				var columvalue= dataset.getColumn(i, columid);
223 				var checklist = (""+validlist[j].value).split(",");
224 				var nullcheck = validlist[j].notnull;
225 
226 				// 필수입력 체크
227 				if (nullcheck == true)
228 				{
229 					if (this.gfnIsNull(columvalue))
230 					{
231 						// Validation ERROR class 사용 시
232 						if (this.useErrorClass) {
233 							this.gfnResetValidationCss(arguments);
234 						}
235 
236 						// row 이동 및 focus 저장
237 						dataset.set_rowposition(i);
238 						this.validationObject = arguments[m];
239 						this.validationColumn = columid;
240 
241 						var titile = "";
242 						if (columname.indexOf("^_^") != -1) {
243 							titile = columname.split("^_^")[0];
244 						}
245 						else {
246 							titile = columname;
247 						}
248 
249 						// 메시지 처리 : ${} 필수 입력 항목입니다.
250 						//this.gfnAlert("msg.err.validator", [this.gfnGetKoreanTarget(titile, "은(는)")+" 필수 입력 항목입니다."], null, "gfnValidationCallback");
251 						// 메시지 처리 : {0} 은(는) 필수 입력 항목입니다.
252 						this.gfnAlert("msg.err.validator.required", [titile], null, "gfnValidationCallback");
253 						
254 						return false;
255 					}
256 				}
257 
258 				// 필수값이 아닌 경우는 체크할 값이 null이 아닌 경우에만 체크
259 				if (!this.gfnIsNull(columvalue))
260 				{
261 					// Rule에 따른 정합성 체크
262 					for(var k=0;k<checklist.length;k++)
263 					{
264 						var msg = "";
265 
266 						// 컬럼의 두개의 값을 이용해서 처리하는 check 대상
267 						var checkrule =  checklist[k].split(":")[0];
268 						if (checkrule == "equalto" || checkrule == "fromto" || checkrule == "comparebig" || checkrule == "comparesmall")
269 						{
270 							var compare1;
271 							var compare2;
272 
273 							if( checklist[k].split(":").length == 3 ) {
274 								compare1 = dataset.getColumn(i, checklist[k].split(":")[1]);
275 								compare2 = dataset.getColumn(i, checklist[k].split(":")[2]);
276 							}
277 							else {
278 								compare1 = dataset.getColumn(i, columid);
279 								compare2 = dataset.getColumn(i, checklist[k].split(":")[1]);
280 							}
281 							msg = this.gfnValidationCheckRule(columname, columvalue, checklist[k], compare1, compare2);
282 						}
283 						// 컬럼의 한개 값을 이용해서 처리하는 check 대상
284 						else {
285 							msg = this.gfnValidationCheckRule(columname, columvalue, checklist[k]);
286 						}
287 
288 						// 에러시
289 						if (!this.gfnIsNull(msg))
290 						{
291 							// Validation ERROR class 사용 시
292 							if (this.useErrorClass) {
293 								this.gfnResetValidationCss(arguments);
294 							}
295 
296 							// row 이동 및 focus 저장
297 							dataset.set_rowposition(i);
298 							this.validationObject = arguments[m];
299 							this.validationColumn = columid;
300 
301 							// 메시지 처리 : ${}
302 							this.gfnAlert("msg.err.validator", [msg], null, "gfnValidationCallback");
303 							
304 							return false;
305 						}
306 					}  // Rule에 따른 정합성 체크
307 				}  // 필수값이 아닌 경우는 체크할 값이 null이 아닌 경우에만 체크
308 			}  // Rule 만큼 처리
309 		}  // 데이터셋의 row 만큼 정합성 체크
310 	}  // arguments로 넘어온 체크 대상 만큼 정합성 체크
311 
312 	// Validation ERROR class 사용 시 cssclass 원복
313 	if (this.useErrorClass) {
314 		this.gfnResetValidationCss(arguments);
315 	}
316 
317 	// 정합성 체크 건수 확인
318 	if (updatecount == 0)
319 	{
320 		trace("정합성 체크 대상이 없어 Validation을 체크가 되지 않았습니다.");
321 		return false;
322 	}
323 
324 	return true;
325 };

327 /**
328  * @class 데이터의 정합성을 체크하여 그 결과를 리턴한다. 메세지가 "" 널이면.. 정상 널이 아니면 실패이다.
329  * @param {String} itemName - Column title
330  * @param {Stirng} itemValue - Column 값
331  * @param {String} validationRuleSet - validation Rule
332  * @return {Stirng} 정상이면 "" 실패이면 "XXX는 7자리입니다." 와 같은 완성형 에러메세지
333  */
334 this.gfnValidationCheckRule = function(itemName, itemValue, validationRuleSet, compare1, compare2)
335 {
336 	var titile     = "";
337 	var columname1 = "";
338 	var columname2 = "";
339 	if (itemName.indexOf("^_^") != -1) {
340 		titile     = itemName.split("^_^")[0];
341 		columname1 = itemName.split("^_^")[0];
342 		columname2 = itemName.split("^_^")[1];
343 	}
344 	else {
345 		titile = itemName;
346 	}
347 				
348 	var arrItem2  =  validationRuleSet.split(":");
349 	var checkrule =  arrItem2[0];
350 
351 	checkrule = checkrule.toLowerCase();
352 	switch(checkrule)
353 	{
354 		// size 크기 지정 : length:7
355 		case "length":
356 			if( (itemValue+"").length != parseInt(arrItem2[1]))
357 			{
358 				//return this.gfnGetKoreanTarget(titile, "은(는)")+" "+arrItem2[1]+"자리 입니다.";
359 				// {0} 의 입력값은 {1} 자리이어야 합니다.
360 				return this.gfnGetMessage("msg.err.validator.length",[titile, arrItem2[1]]);
361 			}
362 			break;
363 		// size 크기의 범위 : rangelength:2:3
364 		case "rangelength":
365 			if( (itemValue+"").length < parseInt(arrItem2[1]) || (itemValue+"").length > parseInt(arrItem2[2]))
366 			{
367 				//return this.gfnGetKoreanTarget(titile, "은(는)")+" ("+arrItem2[1]+")자리에서 ("+arrItem2[2]+")자리로 등록해야 합니다.";
368 				// {0} 은(는) {1} 와(과) {2} 사이의 자리이어야 합니다.
369 				return this.gfnGetMessage("msg.err.validator.rangelength",[titile, arrItem2[1], arrItem2[2]]);
370 			}
371 			break;
372 		// 최대 size 크기: maxlength:7
373 		case "maxlength":
374 			if (itemValue.length > parseInt(arrItem2[1]))
375 			{
376 				//return this.gfnGetKoreanTarget(titile, "은(는)")+" 최대 "+arrItem2[1]+"자리 입니다.";
377 				// {0} 의 입력값의 길이는 {1} 이하이어야 합니다.
378 				return this.gfnGetMessage("msg.err.validator.maxlength",[titile, arrItem2[1]]);
379 			}
380 			break;
381 		// 최소 size 크기: minlength:7
382 		case "minlength":
383 			if (itemValue.length < parseInt(arrItem2[1]))
384 			{
385 				//return this.gfnGetKoreanTarget(titile, "은(는)")+" 최소 "+arrItem2[1]+"자리 이상이어야 합니다.";
386 				// {0} 의 입력값의 길이는 {1} 이상이어야 합니다.
387 				return this.gfnGetMessage("msg.err.validator.minlength",[titile, arrItem2[1]]);
388 			}
389 			break;
390 		// 최대 size 크기(Byte) : maxlengthB:3
391 		case "maxlengthbyte":
392 			if (this.lookupFunc("gfnLengthByte").call(itemValue) > parseInt(arrItem2[1]))
393 			{
394 				//return this.gfnGetKoreanTarget(titile, "은(는)")+" 최대 "+arrItem2[1]+"자리 입니다.";
395 				// {0} 의 입력값의 길이는 {1} 이하이어야 합니다.
396 				return this.gfnGetMessage("msg.err.validator.maxlength",[titile, arrItem2[1]]);
397 			}
398 			break;
399 		// 최소 size 크기(Byte) : minlengthB:3
400 		case "minlengthbyte":
401 			if (this.lookupFunc("gfnLengthByte").call(itemValue) < parseInt(arrItem2[1]))
402 			{
403 				//return this.gfnGetKoreanTarget(titile, "은(는)")+" 최소 "+arrItem2[1]+"자리 이상이어야 합니다.";
404 				// {0} 의 입력값의 길이는 {1} 이상이어야 합니다.
405 				return this.gfnGetMessage("msg.err.validator.minlength",[titile, arrItem2[1]]);
406 			}
407 			break;
408 		// 숫자 여부 : digits
409 		case "digits":		
410 			if (!this.lookupFunc("gfnIsDigit").call(itemValue))
411 			{
412 				//return this.gfnGetKoreanTarget(titile, "은(는)")+" 숫자만 입력 가능합니다.";
413 				// {0} 은(는) 숫자만 입력 가능합니다.
414 				return this.gfnGetMessage("msg.err.validator.digits",[titile]);
415 			}
416 			break;
417 		// 해당 숫자 이하 : min:7
418 		case "min":
419 			if (parseFloat(itemValue) < parseFloat(arrItem2[1]))
420 			{
421 				//return this.gfnGetKoreanTarget(titile, "은(는)")+" "+arrItem2[1]+" 이상의 숫자만 입력 가능합니다.";
422 				// {0} 은(는) {1} 이상의 숫자만 입력 가능합니다.
423 				return this.gfnGetMessage("msg.err.validator.min",[titile, arrItem2[1]]);
424 			}
425 			break;
426 		// 해당 숫자 이상 : max:7
427 		case "max":
428 			if (parseFloat(itemValue) > parseFloat(arrItem2[1]))
429 			{
430 				//return this.gfnGetKoreanTarget(titile, "은(는)")+" "+arrItem2[1]+" 이하의 숫자만 입력 가능합니다.";
431 				// {0} 은(는) {1} 이하의 숫자만 입력 가능합니다.
432 				return this.gfnGetMessage("msg.err.validator.max",[titile, arrItem2[1]]);
433 			}
434 			break;
435 		// 소숫점 자리수 비교 - declimit:3
436 		case "declimit":
437 			var isExistDot = itemValue.indexOf(".");
438 			if (isExistDot == -1)
439 			{
440 				//return this.gfnGetKoreanTarget(titile, "은(는)")+" 소숫점 "+arrItem2[1]+" 자리로 구성되어야 합니다.";
441 				// {0} 은(는) 소숫점 {1} 자리로 구성되어야 합니다.
442 				return this.gfnGetMessage("msg.err.validator.declimit",[titile, arrItem2[1]]);
443 			}
444 			else
445 			{
446 				var decLen = itemValue.substr(isExistDot + 1, itemValue.length);
447 				if (decLen.length != parseInt(arrItem2[1]))
448 				{
449 					//return this.gfnGetKoreanTarget(titile, "은(는)")+" 소숫점 "+arrItem2[1]+" 자리로 구성되어야 합니다.";
450 					// {0} 은(는) 소숫점 {1} 자리로 구성되어야 합니다.
451 					return this.gfnGetMessage("msg.err.validator.declimit",[titile, arrItem2[1]]);
452 				}
453 			}
454 			break;
455 		// 날짜 년월일 체크 : date
456 		case "date":
457 			if (!this.lookupFunc("gfnIsYMD").call(itemValue))
458 			{
459 				//return titile + "에 등록된 날짜가 잘못되었습니다.";
460 				// {0} 은(는) 유효하지 않은 날짜 형식입니다.
461 				return this.gfnGetMessage("msg.err.validator.date",[titile]);
462 			}
463 			break;
464 		// 날짜 년월 체크 : dateym
465 		case "dateym":
466 			if (!this.lookupFunc("gfnIsYM").call(itemValue))
467 			{
468 				//return titile + "에 등록된 년월 형식의 날짜가 잘못되었습니다.";
469 				// {0} 은(는) 유효하지 않은 년월 형식입니다.
470 				return this.gfnGetMessage("msg.err.validator.dateym",[titile]);
471 			}
472 			break;
473 		// 사이의 값인지 비교 - range:40:100
474 		case "range":
475 			if (parseInt(itemValue) < parseInt(arrItem2[1]) || parseInt(itemValue) > parseInt(arrItem2[2]))

808 /**
809  * @class      	문자열(한글)의 초성/중성/종성의 정보를 가져온다.
810  * @return 		{Array} 초성/중성/종성의 갯수만큼을 Array 배열로 리턴한다.
811  */
812 String.prototype.toKorChars = function()
813 {
814     var cCho  = [ 'ㄱ', 'ㄲ', 'ㄴ', 'ㄷ', 'ㄸ', 'ㄹ', 'ㅁ', 'ㅂ', 'ㅃ', 'ㅅ', 'ㅆ', 'ㅇ', 'ㅈ', 'ㅉ', 'ㅊ', 'ㅋ', 'ㅌ', 'ㅍ', 'ㅎ' ],
815         cJung = [ 'ㅏ', 'ㅐ', 'ㅑ', 'ㅒ', 'ㅓ', 'ㅔ', 'ㅕ', 'ㅖ', 'ㅗ', 'ㅘ', 'ㅙ', 'ㅚ', 'ㅛ', 'ㅜ', 'ㅝ', 'ㅞ', 'ㅟ', 'ㅠ', 'ㅡ', 'ㅢ', 'ㅣ' ],
816         cJong = [ '', 'ㄱ', 'ㄲ', 'ㄳ', 'ㄴ', 'ㄵ', 'ㄶ', 'ㄷ', 'ㄹ', 'ㄺ', 'ㄻ', 'ㄼ', 'ㄽ', 'ㄾ', 'ㄿ', 'ㅀ', 'ㅁ', 'ㅂ', 'ㅄ', 'ㅅ', 'ㅆ', 'ㅇ', 'ㅈ', 'ㅊ', 'ㅋ', 'ㅌ', 'ㅍ', 'ㅎ' ],
817         cho, jung, jong;
818 
819     var str = this,
820         cnt = str.length,
821         chars = [],
822         cCode;
823 
824     for (var i = 0; i < cnt; i++) 
825 	{
826         cCode = str.charCodeAt(i);
827 
828 		if (cCode == 32) { continue; }
829 
830         // 한글이 아닌 경우
831         if (cCode < 0xAC00 || cCode > 0xD7A3) {
832             chars.push(str.charAt(i));
833             continue;
834         }
835 
836         cCode  = str.charCodeAt(i) - 0xAC00;
837 
838         jong = cCode % 28; // 종성
839         jung = ((cCode - jong) / 28 ) % 21 // 중성
840         cho  = (((cCode - jong) / 28 ) - jung ) / 21 // 초성
841 
842         chars.push(cCho[cho], cJung[jung]);
843 		if (cJong[jong] !== '') { chars.push(cJong[jong]); }
844     }
845 
846     return chars;
847 };
