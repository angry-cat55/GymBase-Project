 18 /**
 19  * @class 메세지팝업오픈
 20  * @param {String} sMsgId - 메세지ID	
 21  * @param {Array} arrArg - 메세지에 치환될 부분은 "{0~N}"이 되고 치환값은 배열로 넘김 
 22  * @param {String} [sPopId] - 팝업ID(하나의 callback함수에서 중복된 메시지 처리를 할 경우 PopId구분을 위해 unique한 ID 반드시 사용)
 23  * @param {String} [sCallback] - 팝업콜백 (confirm성 메시지를 사용시 반드시 필요)
 24  * @return N/A
 25  * @example
 26  * this.gfnAlert(this, "A", "확인하세요");	
 27  */
 28 this.gfnAlert = function (sMsgId, arrArg, sPopId, sCallback)
 29 {
 30     var objApp = this.gfnGetApplication();
 31 	if(objApp.gdsMessage.findRow("msgId", sMsgId) < 0) return false;
 32 
 33 	// 다국어 처리
 34 	var sNowLang = nexacro.getEnvironmentVariable("evLanguage");
 35 	var sColumn  = "msgText";
 36 	if (sNowLang != "KO") {
 37 		sColumn = sColumn + sNowLang;
 38 	}
 39 	
 40 	var sMsg = objApp.gdsMessage.lookup("msgId", sMsgId, sColumn);
 41 
 42 	if( this.gfnIsNull(sMsg) ) sMsg = "확인";
 43 	// 줄바꿈 변경
 44 	sMsg = sMsg.replace(/\\n/g, String.fromCharCode(10));
 45 	sMsg =  this.gfnConvertMessage(sMsg, arrArg);
 46 	
 47 	var sMsgType = objApp.gdsMessage.lookup("msgId", sMsgId, "msgType");	
 48 	if( this.gfnIsNull(sPopId) ) sPopId = sMsgId;
 49 	
 50 	var sMsgUrl ="";
 51 	switch(sMsgType) {
 52 		case "A":
 53 			sMsgUrl = "cmm::cmmAlert.xfdl";
 54 			break;
 55 		case "C":
 56 			sMsgUrl = "cmm::cmmConfirm.xfdl";
 57 			if(this.gfnIsNull(sCallback)) trace("callback함수를 지정하지 않았습니다");
 58 			break;
 59 	}
 60 	
 61 	var oArg = {paramContents:sMsg};
 62 	var oOption = {titlebar:"false"};	
 63 	
 64 	// messagePopup
 65 	if (nexacro.getEnvironmentVariable("evMessagePopup") == "true") {
 66 		this.gfnOpenPopup(sPopId,sMsgUrl,oArg,sCallback,oOption);
 67 	}
 68 	// alert-cofirm
 69 	else {
 70 		if (sMsgType == "A") {
 71 			alert(sMsg);
 72 		}
 73 		else {
 74 			confirm(sMsg);
 75 		}
 76 	}
 77 };

 79 /**
 80  * @class 메세지 치환
 81  * @param {String} msg - 메세지	
 82  * @param {Array} values - 메세지에 치환될 부분은 "{0~N}"이 되고 치환값은 배열로 넘김 
 83  * @return {String}
 84  */
 85 this.gfnConvertMessage = function(msg, values) 
 86 {
 87     return msg.replace(/\{(\d+)\}/g, function() {
 88         return values[arguments[1]];
 89     });
 90 };

 96  * @return {String}
