 17 /**
 18  * @class excel export <br>
 19  * @param {Object} objGrid - Grid Object	
 20  * @param {String} [sSheetName]	- sheet name
 21  * @param {String} [sFileName]	- file name
 22  * @return N/A
 23  * @example
 24  * this.gfnExcelExport(this.grid_export, "SheetName","");
 25  */
 26 this.gfnExcelExport = function(objGrid,  sSheetName, sFileName)
 27 {
 28 
 29 	this.setWaitCursor(true);
 30 	var objGrid_excel = objGrid;
 31 	
 32 	var regExp = /[?*:\/\[\]]/g;  				//(엑셀에서 지원하지않는 모든 문자)
 33 	
 34 	sFileName = sFileName.replace(regExp,"");	//파일명에 특수문자 제거
 35 	sSheetName = sSheetName.replace(regExp,""); //시트명에 특수문자 제거
 36 	
 37 	//fileName nullcheck
 38 	sFileName = this.gfnIsNull(sFileName) ? this.gfnGetArgument("menuNm")+"_"+this.gfnGetDate() : sFileName;
 39 	//sheetName nullcheck
 40 	sSheetName = this.gfnIsNull(sSheetName) ? "sheet1" : sSheetName;
 41 	//sheetName 30이상일경우 기본시트명
 42 	if( String(sSheetName).length > 30 ){
 43 		sSheetName =  "sheet1";
 44 	}
 45 	
 46 	var svcUrl = "svcUrl::XExportImport.do";
 47 	this.objExport = null
 48 	this.objExport = new ExcelExportObject();
 49 	
 50 	this.objExport.objgrid = objGrid_excel;
 51 	this.objExport.set_exporturl(svcUrl);
 52 	this.objExport.addExportItem(nexacro.ExportItemTypes.GRID, objGrid_excel, sSheetName+"!A1","allband","allrecord");
 53 	this.objExport.set_exportfilename(sFileName);	
 54 	
 55  	this.objExport.set_exporteventtype("itemrecord");
 56  	this.objExport.set_exportuitype("none");
 57  	this.objExport.set_exportmessageprocess("");
 58 	this.objExport.addEventHandler("onsuccess", this.gfnExportOnsuccess, this);	
 59 	this.objExport.addEventHandler("onerror", this.gfnExportOnerror, this);	
 60 		
 61 	var result = this.objExport.exportData();
 62 };

 89 /**
 90  * @class  excel import( 데이터 헤더포함 ) <br>
 91  * @param {String} objDs - dataset	
 92  * @param {String} [sSheet]	- sheet name(default:Sheet1)
 93  * @param {String} sHead - Head 영역지정	
 94  * @param {String} [sBody] - body 영역지정(default A2)	
 95  * @param {String} [sCallback]	- callback 함수
 96  * @param {String} [sImportId] - import id(callback호출시 필수)	
 97  * @param {Object} [objForm] - form object(callback호출시 필수)
 98  * @return N/A
 99  * @example
100  * this.gfnExcelImportAll("dsList","SheetName","A1:G1","A2","fnImportCallback","import",this);
101  */
102 this.gfnExcelImportAll = function(objDs,sSheet,sHead,sBody,sCallback,sImportId,objForm)
103 {	
104 	this.setWaitCursor(true);    	
105 	
106 	if(this.gfnIsNull(sSheet)) sSheet = "sheet1";
107 	if(this.gfnIsNull(sBody)) sBody = "A2";
108 	if(this.gfnIsNull(sHead)) return false;
109 	
110 	var svcUrl = "svcUrl::XExportImport.do";
111 	
112 	var objImport ;	
113 	
114 	objImport = new nexacro.ExcelImportObject(objDs+"_ExcelImport",this);				
115 	objImport.set_importurl(svcUrl);						
116 	objImport.set_importtype(nexacro.ImportTypes.EXCEL);			
117 	
118 	if (!this.gfnIsNull(sCallback))
119 	{
120 		objImport.callback = sCallback;
121 		objImport.importid = sImportId;
122 		objImport.form = objForm;
123 	}
124 	
125 	objImport.addEventHandler("onsuccess", this.gfnImportAllOnsuccess, this);
126 	objImport.addEventHandler("onerror", this.gfnImportAllOnerror, this);	
127 	var sParam1 = "[Command=getsheetdata;Output=outds;Head="+sSheet+"!"+sHead+";Body="+sSheet+"!"+sBody+"]";
128 	var sParam2 = "["+objDs+"=outds]";
129 
130 
131 	objImport.importData("", sParam1, sParam2);						
132 	objImport = null;	 
133 };

154 /**
155  * @class  excel import( 데이터 헤더제외 ) <br>
156  * @param {String} sDataset - dataset	
157  * @param {String} [sSheet]	- sheet name
158  * @param {String} [sBody] - body 영역지정	
159  * @param {String} [sCallback] - callback 함수	
160  * @param {String} [sImportId] - import id(callback호출시 필수)	
161  * @param {Object} [objForm] - form object(callback호출시 필수)	
162  * @return N/A
163  * @example
164  * this.gfnExcelImport("dsList","SheetName","A2","fnImportCallback","import",this);
165  */
166 this.gfnExcelImport = function(sDataset, sSheet, sBody, sCallback, sImportId, objForm)
167 {
168 	this.setWaitCursor(true);    	
169 	
170 	if(this.gfnIsNull(sSheet)) sSheet = "sheet1";
171 	if(this.gfnIsNull(sBody)) sBody = "A2";
172 	
173 	var svcUrl = "svcUrl::XExportImport.do";
174 	
175 	var objImport;	
176 	objImport = new nexacro.ExcelImportObject(sDataset+"_ExcelImport",this);				
177 	objImport.set_importurl(svcUrl);						
178 	objImport.set_importtype(nexacro.ImportTypes.EXCEL);			
179 	objImport.outds = sDataset;
180 
181 	if (!this.gfnIsNull(sCallback))
182 	{
183 		objImport.callback = sCallback;
184 		objImport.importid = sImportId;
185 		objImport.form = objForm;
186 	}
187 	
188 	//out dataset 생성(차후 onsucess 함수에서 헤더생성하기 위한)
189 	var sOutDsName = sDataset+"_outds";	
190 	if(this.isValidObject(sOutDsName)) this.removeChild(sOutDsName);			
191 	var objOutDs = new Dataset();
192 	objOutDs.name = sOutDsName;
193 	this.addChild(objOutDs.name, objOutDs);
194 	
195 	objImport.addEventHandler("onsuccess", this.gfnImportOnsuccess, this);
196 	objImport.addEventHandler("onerror", this.gfnImportAllOnerror, this);	
197 	var sParam = "[command=getsheetdata;output=outDs;body=" + sSheet + "!" + sBody +";]";
198  	var sParam2 = "[" + sOutDsName + "=outDs]";
199  	
200 	objImport.importData("", sParam, sParam2);						
201 	objImport = null;	
202 	
203 	this.setWaitCursor(false);
204 };

206 /**
207  * @class excel import on success <br>
208  * @param {Object} obj	
209  * @param {Event} e		
210  * @return N/A
211  * @example
212  */
213 this.gfnImportOnsuccess = function(obj,  e)
214 {		
215 	this.setWaitCursor(false);
216 	
217 	var objOutDs = this.objects[obj.outds+"_outds"];
218 	var objOrgDs = this.objects[obj.outds];
219 	var sCallback = obj.callback;
220 	var sImportId = obj.importid;
221 	var objForm = obj.form;
222 	var sColumnId;
223 
224 	//기존 데이터셋의 내용으로 헤더복사
225 	for (var i=0; i<objOrgDs.getColCount(); i++)
226 	{
227 		sColumnId = "Column"+i;
228 		if (sColumnId != objOrgDs.getColID(i))
229 		{
230 			objOutDs.updateColID(sColumnId, objOrgDs.getColID(i))
231 		}
232 	}
233 	
234 	objOrgDs.clearData();
235 	objOrgDs.copyData(objOutDs);
236 
237 	//화면의 callback 함수 호출
238 	if (!this.gfnIsNull(sCallback)) {
239 		if (this[sCallback]) this.lookupFunc(sCallback).call(sImportId);
240 	}
241 };
