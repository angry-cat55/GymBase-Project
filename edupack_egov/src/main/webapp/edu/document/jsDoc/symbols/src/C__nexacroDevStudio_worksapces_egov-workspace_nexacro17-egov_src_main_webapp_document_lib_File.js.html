 17 /**
 18  * @class 현재 Form 상의 FileUpload 컴포넌트를 서버에 업로드한다. <br>
 19  * @param {Object} objFileUpload - 파일업로드 컴포넌트
 20  * @param {String} [sUrl] - 파일업로드 서비스 호출 경로
 21  * @param {String} [sPath] - 파일업로드시킬 폴더 위치
 22  * @return N/A
 23  * @example 
 24  * this.gfnFileUpload(objFileUpload);
 25  */
 26 this.gfnFileUpload = function(objFileUpload, sUrl, sPath)
 27 {	
 28 	var svcUrl = this.gfnGetServerUrl();
 29 	
 30 	if (this.gfnIsNull(sUrl)) sUrl = svcUrl;
 31 	
 32 	//파일업로드 서비스 호출 경로
 33 	var sFileUrl = sUrl + "fileUpload.jsp";
 34     
 35 	//파일 업로드 시킬 폴더 위치 지정
 36 	if (this.gfnIsNull(sPath)) sPath = "PATH=upload";
 37 	
 38 	var bSucc = objFileUpload.upload(sFileUrl + "?" + sPath);
 39 	trace("bSucc >> " + bSucc);
 40 };

 42 /**
 43  * @class 현재 Form 상의 FileDownload 컴포넌트를 이용하여 지정한 위치에서 원하는 파일을 다운로드한다. <br>
 44  * @param {Object} objFileDownload - 파일다운로드 컴포넌트
 45  * @param {String} sFilename - 다운로드 할 파일명
 46  * @param {String} [sUrl] - 파일업로드 서비스 호출 경로
 47  * @param {String} [sPath] - 파일업로드시킬 폴더 위치
 48  * @return N/A
 49  * @example this.gfnFileUpload(objFileUpload, sFilename);
 50  */
 51 this.gfnFileDownload = function(objFileDownload, sFilename, sUrl, sPath)
 52 {
 53 	var svcUrl = this.gfnGetServerUrl();
 54 	if (this.gfnIsNull(sUrl)) sUrl = svcUrl;
 55 	
 56 	
 57 	//파일다운로드 서비스 호출 경로
 58 	var sFileUrl = sUrl + "fileDownload.jsp";
 59 	
 60 	//파일 다운로드할 폴더 위치 지정
 61 	if (this.gfnIsNull(sPath)) sPath = "PATH=upload";
 62 	
 63 	objFileDownload.download(sFileUrl + "?" + sPath + "&file=" + sFilename);
 64 };

 66 /**
 67  * @class File Path 문자열(예 : C:\a\b\filename.ext)에서 File명(예 : filename)을 추출 <br>
 68  * @param {String} sPath - File Path 문자열 (예 : "C:\a\b\filename.ext")
 69  * @param {String} bExt - extend를 return되는 File명에 포함시킬지 여부 ( 옵션 : Default=false )
 70  * @return {String} 
 71  * 성공 : <br>
 72  * bExt가 true인 경우 ==> sPath에서 File명(예 : "filename.ext") <br>
 73  * bExt가 false인 경우 ==> sPath에서 File명(예 : "filename") <br>
 74  * 실패 : "" <br>
 75  */
 76 this.gfnGetFileName = function (sPath, bExt)
 77 {
 78 	var start_pos,end_pos,tmp_pos,filename;
 79 
 80 	if (this.gfnIsNull(sPath)) 
 81 	{
 82 		return "";
 83 	}
 84 	if (this.gfnIsNull(bExt)) 
 85 	{
 86 		bExt = false;
 87 	}
 88 
 89 	start_pos = Math.max(this.gfnPosReverse(sPath, "\\"), this.gfnPosReverse(sPath, "/"));
 90 	tmp_pos = this.gfnPosReverse(sPath, "::");
 91 	if (tmp_pos > 0) 
 92 	{
 93 		tmp_pos++;
 94 	}
 95 	start_pos = Math.max(start_pos, tmp_pos);
 96 	if (bExt == false) 
 97 	{
 98 		end_pos = this.gfnPosReverse(sPath, ".");
 99 		if (end_pos < 0) 
100 		{
101 			end_pos = sPath.length;
102 		}
103 		filename = sPath.substr(start_pos + 1, end_pos - start_pos - 1);
104 	}
105 	else 
106 	{
107 		filename = sPath.substr(start_pos + 1);
108 	}
109 
110 	return filename;
111 };
