 16 /**
 17  * @class 서비스 호출 공통함수 <br>
 18  * Dataset의 값을 갱신하기 위한 서비스를 호출하고, 트랜젝션이 완료되면 콜백함수을 수행하는 함수
 19  * @param {String} strSvcId - 서비스 ID
 20  * @param {String} strSvcUrl - 서비스 호출 URL 
 21  * @param {String} [inData]	- input Dataset list("입력ID=DataSet ID" 형식으로 설정하며 빈칸으로 구분)
 22  * @param {String} [outData] - output Dataset list("DataSet ID=출력ID" 형식으로 설정하며 빈칸으로 구분)
 23  * @param {String} [strArg]	- 서비스 호출시 Agrgument
 24  * @param {String} [callBackFnc] - 콜백 함수명
 25  * @param {Boolean} [isAsync] - 비동기통신 여부 
 26  * @return N/A
 27  * @example
 28  * var strSvcUrl = "transactionSaveTest.do";
 29  * var inData    = "dsList=dsList:U";
 30  * var outData   = "dsList=dsList";
 31  * var strArg    = "";
 32  * this.gfnTransaction("save", strSvcUrl, inData, outData, strArg, "fnCallback", true);
 33  */ 
 34 this.gfnTransaction = function(strSvcId, strSvcUrl, inData, outData, strArg, callBackFnc, isAsync)
 35 {
 36 	if (this.gfnIsNull(strSvcId) || this.gfnIsNull(strSvcUrl))
 37 	{
 38 		trace("Error : gfnTransaction() 함수의 인자값이 부족합니다.");
 39 		return false;
 40 	}
 41 	
 42 	// fnCallback 함수 기본값 설정
 43 	if (this.gfnIsNull(callBackFnc)) callBackFnc = "fnCallback";
 44 	
 45 	var objDate = new Date();
 46 	var nStartTime = objDate.getTime();
 47     var sStartDate = objDate.getYear()
 48 						+"-"+String(objDate.getMonth()).padLeft(2, '0')
 49 						+"-"+String(objDate.getDate()).padLeft(2, '0')
 50 						+" "+String(objDate.getHours()).padLeft(2, '0')
 51 						+":"+String(objDate.getMinutes()).padLeft(2, '0')
 52 						+":"+String(objDate.getSeconds()).padLeft(2, '0')
 53 						+" "+objDate.getMilliseconds();
 54 
 55 	// Async
 56 	if ((isAsync != true) && (isAsync != false)) isAsync = true;	
 57 	
 58 	// 1. callback에서 처리할 서비스 정보 저장
 59 	var objSvcID = { 
 60 			svcId     : strSvcId,
 61 			svcUrl    : strSvcUrl,
 62 			callback  : callBackFnc,
 63 			isAsync   : isAsync,
 64 			startDate : sStartDate,
 65 			startTime : nStartTime };
 66 	
 67 	// 2. strServiceUrl
 68 	var strServiceUrl = "svcUrl::" + strSvcUrl;
 69 	
 70 	// 3. strArg
 71 	var strArguments = "";
 72 	if (this.gfnIsNull(strArg)) {
 73 		strArguments = "";
 74 	}
 75 	else { 
 76 		strArguments = strArg;
 77 	}
 78 
 79 	// 개발시에는 xml, 개발서버/운영서버는 SSV로 통신
 80 	var nDataType;	
 81 	if (nexacro.getEnvironmentVariable("evRunMode") == "2") 
 82 	{
 83 		nDataType = 2;
 84 	}
 85 	else {
 86 		nDataType = 0;
 87 	}
 88 	
 89 	this.transaction( JSON.stringify(objSvcID)  //1.svcID
 90 					, strServiceUrl             //2.strServiceUrl
 91 					, inData                    //3.inDataSet
 92 					, outData                   //4.outDataSet
 93 					, strArguments              //5.arguments
 94 					, "gfnCallback"				//6.strCallbackFunc
 95 					, isAsync                   //7.bAsync
 96 					, nDataType                 //8.nDataType : 0(XML 타입), 1((Binary 타입),  2(SSV 타입) --> HTML5에서는 Binary 타입은 지원안함
 97 					, false);                   //9.bCompress ( default : false ) 
 98 };

127 		}
128 	}
129 
130 	// 서비스 실행결과 출력
131 	var sStartDate = objSvcID.startDate;
132 	var nStartTime = objSvcID.startTime;
133 	
134 	var objDate = new Date();

136 					+"-"+String(objDate.getMonth()).padLeft(2, '0')
