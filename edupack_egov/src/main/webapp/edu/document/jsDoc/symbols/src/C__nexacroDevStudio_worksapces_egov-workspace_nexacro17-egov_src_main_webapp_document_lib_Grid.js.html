379 /**
380  * @class 정렬가능여부리턴
381  * @param {Object} grid - 대상그리드
382  * @param {Number} headCellIndex - 대상셀INDEX
383  * @param {Boolean}multiple - 멀티소트여부 
384  * @param {Number} sortStatus - 소트상태  
385  * @return{Boolean} sort 가능/불가능 여부
386  * @example
387  * this._gfnGridSetSortStatus(obj, e.cell, multiple);	
388  */
389 this._gfnGridSetSortStatus = function(grid, headCellIndex, isMultiple, sortStatus, bodyCellIndex)
390 {
391 	// head cell index 에 해당하는 body cell index
392 	if( this.gfnIsNull(bodyCellIndex)){
393 		bodyCellIndex = this._gfnGridGetBodyCellIndex(grid, headCellIndex);
394 	}
395 	if ( bodyCellIndex < 0 ) return false;
396 	
397 	// body cell index 에 해당하는 바인드 컬럼명
398 	var columnName = this._gfnGridGetBindColumnNameByIndex(grid, bodyCellIndex);
399 	if ( this.gfnIsNull(columnName) ){
400 		trace("Check Grid body cell bind value");
401 		return false;
402 	}
403 	
404 	if ( this.gfnIsNull(isMultiple) ) isMultiple = false;
405 	if ( this.gfnIsNull(sortStatus) ) sortStatus = -1;
406 	
407 	// 대상 grid 에 정렬정보를 가지는 사용자 속성 확인/추가
408 	if ( this.gfnIsNull(grid.sortInfos) ){
409 		grid.sortInfos = {};
410 	}
411 	
412 	// 정렬대상컬럼 (순서중요)
413 	if ( this.gfnIsNull(grid.sortItems) ){
414 		grid.sortItems = [];
415 	}
416 	
417 	var sortInfos = grid.sortInfos,
418 		sortItems = grid.sortItems,
419 		sortInfo = sortInfos[columnName],
420 		sortItem,
421 		status;
422 	
423 	if ( this.gfnIsNull(sortInfo) )
424 	{
425 		var headText = grid.getCellText(-1, headCellIndex);
426 		
427 		// executeSort에서 정렬 표시를 위해 cell index 가 필요한데
428 		// cell moving 될 경우 index는 변하므로 cell object 를 참조하여 값을 얻어온다. 		
429 		var refCell = this._gfnGridGetGridCellObject(grid, "head", headCellIndex);
430 		sortInfo = sortInfos[columnName] = { status: 0, text: headText, refCell: refCell};
431 	}
432 	// set sort status
433 	if ( isMultiple ) {		
434 		status = sortInfo.status;
435 		if ( sortStatus == -1 ) {
436 			if ( status == 0 ) {
437 				sortInfo.status = 1;
438 			} 
439 			else if ( status == 1 ) {
440 				sortInfo.status = 2;
441 			} 
442 			else if ( status == 2 ) {
443 				sortInfo.status = ( this.SORT_TOGGLE_CANCEL ? 0 : 1);
444 			}
445 		}
446 		else {
447 			sortInfo.status = sortStatus;
448 		}
449 	}else {
450 		for (var p in sortInfos) {
451 			if ( sortInfos.hasOwnProperty(p) )
452 			{
453 				sortInfo = sortInfos[p];
454 				if ( p == columnName ) {
455 					status = sortInfo.status;
456 					if ( sortStatus == -1 ) {
457 						if ( status == 0 ) {
458 							sortInfo.status = 1;
459 						} 
460 						else if ( status == 1 ) {
461 							sortInfo.status = 2;
462 						} 
463 						else if ( status == 2) {
464 							sortInfo.status = ( this.SORT_TOGGLE_CANCEL ? 0 : 1);
465 						}
466 					}else {
467 						sortInfo.status = sortStatus;
468 					}
469 				}else {
470 					sortInfo.status = 0;
471 				}
472 				if ( sortInfo.status == 0 ){
473 					for (var j=0, len2=sortItems.length; j<len2; j++) {
474 						if ( sortItems[j] !== columnName ) {
475 							sortItems.splice(j, 1);
476 							break;
477 						}
478 					}
479 				}
480 			}
481 		}
482 	}
483 	
484 	// 컬럼정보 등록
485 	var hasItem = false;
486 	for (var i=0, len=sortItems.length; i<len; i++) {
487 		if ( sortItems[i] == columnName ) {
488 			hasItem = true;
489 			break;
490 		}
491 	}	
492 	if ( !hasItem ){
493 		sortItems.push(columnName);
494 	}
495 	return true;
496 };

498 /**
499  * @class head cell에 match되는 body cell을 얻어온다
500  * @param {Object}  grid 대상 Grid Component
501  * @param {Number} eadCellIndex head cell index

503  * @example

505  */ 
506 this._gfnGridGetBodyCellIndex = function(grid, headCellIndex, useColspan) 
507 {	//, useColspan) 
508 	if( this.gfnIsNull(useColspan)) useColspan=false;
509 	// Max Head Row Index
510 	var maxHeadRow = 0;
511 	for (var i=0, len=grid.getCellCount("head"); i<len; i++) {
512 		var row = grid.getCellProperty("head", i, "row");
513 		if (maxHeadRow < row) {
514 			maxHeadRow = row;
515 		}
516 	}
517 	// Max Body Row Index
518 	var maxBodyRow = 0;

537 	// Head의 row 가 Body의 row 보다 클 경우 차이 row 를 뺀 것을 대상으로 찾고
538 	// Body의 row 가 Head의 row 보다 크거나 같을 경우 row index가 같은 대상을 찾는다.			
539 	var cellIndex = -1;

560 		cRowspan = parseInt(grid.getCellProperty("body", i, "rowspan"));
561 		if( cRowspan > 1 )
562 		{
563 			if ( useColspan ){
564 				if (sRow >= cRow && nCol <= cCol && cCol < (nCol + nColspan)) 
565 				{		
566 					cellIndex = i;
567 					break;
568 				}		
569 			}else{
570 				if (sRow >= cRow && nCol == cCol && nColspan == cColspan) 
571 				{		
572 					cellIndex = i;
573 					break;
574 				}
575 			}
576 		}else{	
577 			if ( useColspan ){
578 				if (sRow == cRow && nCol <= cCol && cCol < (nCol + nColspan)) 
579 				{		
580 					cellIndex = i;
581 					break;
582 				}		
583 			}else{
584 				if (sRow == cRow && nCol == cCol && nColspan == cColspan) 
585 				{		
586 					cellIndex = i;
587 					break;
588 				}
589 			}
590 		}
591 	}
592 	return cellIndex;

627 /**
628  * @class 소트를 실행한다
629  * @param {Object}  grid 대상 Grid Component
630  * @return{String}  N/A
631  * @example
632  * this._gfnGridExecuteSort(obj);

634 this._gfnGridExecuteSort = function(grid) 
635 {
636 	var sortInfo,

638 		sortInfos = grid.sortInfos,

641 		status,
642 		cell,
643 		sortString = "";
644 		
645 	if ( this.gfnIsNull(sortInfos) || this.gfnIsNull(sortItems) ) return;
646 
647 	// keystring 조합
648 	for (var i=0; i<sortItems.length; i++) {
649 		columnName = sortItems[i];
650 		sortInfo = sortInfos[columnName];
651 		status = sortInfo.status;
652 		cell = sortInfo.refCell;
653 		
654 		// 컬럼삭제 등으로 제거될 수 있으므로 실제 column 이 존재하는지
655 		// 확인하여 없으면 제거해 준다.
656 		if ( this.gfnIsNull(cell) || grid.getBindCellIndex("body", columnName) < 0 ){
657 			// 컬럼정보제거
658 			sortItems.splice(i, 1);
659 			sortInfos[columnName] = null;
660 			delete sortInfos[columnName];
661 			
662 			i--;
663 		}else if ( status > 0 ) {
664 			sortString += (status == 1 ? "+" : "-") + columnName;
665 		}
666 	}
667 	
668 	var ds = grid.getBindDataset();
669 	// keystring 확인
670 	var curKeyString = ds.keystring;
671 	var groupKeyString = "";
672 	
673 	if ( curKeyString.length > 0 && curKeyString.indexOf(",") < 0 ){
674 		var sIndex = curKeyString.indexOf("S:");
675 		var gIndex = curKeyString.indexOf("G:");
676 
677 		if ( sIndex > -1 ){
678 			groupKeyString = "";
679 		}else{
680 			if ( gIndex < 0 )
681 			{
682 				groupKeyString = "G:"+curKeyString;
683 			}
684 			else
685 			{
686 				groupKeyString = curKeyString;
687 			}
688 		}
689 	}else{
690 		var temps = curKeyString.split(",");
691 		var temp;
692 		for (var i=0,len=temps.length; i<len; i++){
693 			temp = temps[i];
694 			if ( temp.length > 0 && temp.indexOf("S:") < 0 ){
695 				if ( temp.indexOf("G:") < 0 )
696 				{
697 					groupKeyString = "G:"+temp;
698 				}else{
699 					groupKeyString = temp;
700 				}
701 			}
702 		}
703 	}
704 	
705 	if ( sortString.length > 0 ){
706 		var sortKeyString = "S:"+sortString;
707 		
708 		if ( groupKeyString.length > 0 ){
709 			ds.set_keystring(sortKeyString + "," + groupKeyString);
710 		}else{
711 			ds.set_keystring(sortKeyString);
712 		}
713 		
714 		grid.sortKeyString = sortKeyString;
715 	}else{		
716 		ds.set_keystring(groupKeyString);
717 		grid.sortKeyString = "";
718 	}
719 
720 	// 정렬표시
721 	var type = this.MARKER_TYPE;
722 	var index, marker;
723 	for (var p in sortInfos) {
724 		if ( sortInfos.hasOwnProperty(p) )
725 		{
726 			sortInfo = sortInfos[p];			
727 			cell = sortInfo.refCell;
728 			if ( cell )
729 			{
730 				index = cell._cellidx;
731 				marker = this.gfnDecode(sortInfo.status, 1, this.MARKER[0], 2, this.MARKER[1], "");
732 				grid.setCellProperty( "head", index, "text", sortInfo.text + marker);
733 			}
734 		}
735 	}
736 };
