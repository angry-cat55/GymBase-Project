 18 /**
 19  * @class  해당 form의 하위 콤포넌트 모두 찾아 다국어 처리
 20  * @param  {Form} objForm - 대상 form
 21  * @return N/A
 22  */
 23 this.gfnInitLang = function (objForm)
 24 {
 25 	var sNowLang    = nexacro.getEnvironmentVariable("evLanguage");			// 현재 언어
 26 	var sChangeLang = nexacro.getEnvironmentVariable("evLanguageChange");	// 바꿀 언어
 27 	
 28 	// Language 변경 전/후 값 확인
 29 	if (sNowLang == "KO" && sNowLang == sChangeLang) return;
 30 
 31 	var arrComp = objForm.components;
 32 	var nLength = arrComp.length;
 33 
 34 	for (var i=0; i<nLength; i++)
 35 	{
 36 		if (arrComp[i] instanceof nexacro.Div)
 37 		{
 38 			this.gfnInitLang(arrComp[i].form); //재귀함수
 39 		}
 40 		else if (arrComp[i] instanceof nexacro.Tab)
 41 		{
 42 			var nPages = arrComp[i].tabpages.length;
 43 
 44 			for (var j=0; j<nPages;j++)
 45 			{
 46 				this.gfnChangeLang(arrComp[i].tabpages[j]);
 47 				this.gfnInitLang(arrComp[i].tabpages[j].form); //재귀함수
 48 			}
 49 		}
 50 		else
 51 		{
 52 			this.gfnChangeLang(arrComp[i]); //다국어설정
 53 		}
 54 	}
 55 	
 56 	// text 사이즈 변경으로 인한 화면 갱신 필요
 57 	objForm.resetScroll();
 58 };

 61 /**
 62  * @class  해당 Component에 대해 다국어 처리
 63  * @param  {Object} objComp - 대상 Object
 64  * @return N/A
 65  */
 66 this.gfnChangeLang = function (objComp)
 67 {
 68 	var sWord = "";		// 용어 code
 69 	var sVal  = "";		// 치환될 값
 70 	
 71 	// Button, Static, CheckBox, Tabpage
 72 	if (objComp instanceof nexacro.Button || objComp instanceof nexacro.Static || objComp instanceof nexacro.CheckBox || objComp instanceof nexacro.Tabpage)
 73 	{
 74 		sWord = objComp.uWord;
 75 		if (this.gfnIsNull(sWord) || this.gfnIsNull(objComp.text)) return;
 76 	
 77 		sVal = this.gfnGetWord(sWord);
 78 		objComp.set_text(sVal);
 79 	}
 80 	// ImageViewer 는 해당 프로젝트의 디자인 파일의 Naming Rule에 맞게 수정 필요
 81 	else if (objComp instanceof nexacro.ImageViewer)
 82 	{		
 83 		sWord = objComp.uWord; 		// image의 경우 용어 code에 true 설정
 84 		if (this.gfnIsNull(sWord) || sWord != "true") return;
 85 
 86 		var sImage   = objComp.image;	// image
 87 		// 이미지 확장자
 88 		var buff     = sImage.split(".");
 89 		var fileExt  = buff[buff.length-1].replace("\")", "");
 90 			
 91 		var sNowLang    = nexacro.getEnvironmentVariable("evLanguage");			// 현재 언어
 92 		var sChangeLang = nexacro.getEnvironmentVariable("evLanguageChange");	// 바꿀 언어
 93 		if (sNowLang != "KO" && sChangeLang == "KO")
 94 		{
 95 			sImage = sImage.replace("_"+sNowLang+"."+fileExt,"."+fileExt);
 96 		}
 97 		else {		
 98 			var nPos = this.gfnPosReverse(sImage,"."+fileExt);
 99 			sImage = sImage.substr(0, nPos) + "_" + nexacro.getEnvironmentVariable("evLanguageChange")+"."+fileExt+"\")";
100 		}
101 		objComp.set_image(sImage);		
102 	}
103 	// Grid
104 	else if (objComp instanceof nexacro.Grid)
105 	{
106 		objComp.set_enableevent(false);
107 		var nCol = objComp.getCellCount("head");
108 
109 		for (var j=0; j<nCol; j++)
110 		{
111 			// Grid의 경우 용어 code를 Header의 expandchar Property에 기술
112 			sWord = objComp.getCellProperty("Head", j, "expandchar");	// 용어 code
113 			if (this.gfnIsNull(sWord)) continue;
114 		
115 			sVal = this.gfnGetWord(sWord);
116 			objComp.setCellProperty("Head", j, "text", String(sVal))
117 		}
118 		objComp.set_enableevent(true);
119 	}	
120 };

122 /**
123  * @class  다국어 처리를 위한 용어 검색
124  * @param  {String} sTargetVal - 검색할 용어
125  * @return {String} 변경할 용어
126  */
127 this.gfnGetWord = function (sWord)
128 {
129 	var objApp 	  = this.gfnGetApplication();
130 	var objLangDs = objApp.gdsWord;
131 	
132 	var sChangeLang = nexacro.getEnvironmentVariable("evLanguageChange");	// 바꿀 언어
133 
134 	var sVal = sWord;
135 	var nRow = objLangDs.findRow("word",sWord);
136 	if (nRow != -1)
137 	{
138 		sVal = objLangDs.getColumn(nRow, sChangeLang);
139 	}
140 	return sVal;
141 };
