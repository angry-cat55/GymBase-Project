 18 /**
 19 * @class frame open <br>
 20 * @param {Object} obj - 화면
 21 * @return N/A
 22 * @example 
 23 * this.gfnFormOnLoad(this);
 24 */
 25 this.gfnFormOnLoad = function(objForm)
 26 {
 27 	var arrComp = objForm.components;
 28 	var nLength = arrComp.length;
 29 
 30 	for (var i=0; i<nLength; i++)
 31 	{
 32 		if (arrComp[i] instanceof nexacro.Div)
 33 		{
 34 			// URL로 링크된 경우에는 존재하는 경우에는 해당 링크된 Form Onload에서 처리하도록 한다.
 35 			if (this.gfnIsNull(arrComp[i].url)) this.gfnFormOnLoad(arrComp[i].form);
 36 		}
 37 		else if (arrComp[i] instanceof nexacro.Tab)
 38 		{
 39 			var nPages = arrComp[i].tabpages.length;
 40 			
 41 			for (var j=0; j<nPages;j++)
 42 			{	
 43 				// URL로 링크된 경우에는 존재하는 경우에는 해당 링크된 Form Onload에서 처리하도록 한다.
 44 				if (this.gfnIsNull(arrComp[i].tabpages[j].url)) this.gfnFormOnLoad(arrComp[i].tabpages[j].form);
 45 			}
 46 		}
 47 		else
 48 		{
 49 			if (arrComp[i] instanceof nexacro.Grid) {
 50 				this.gfnSetGrid(arrComp[i]);
 51 			}
 52 			
 53 			if (arrComp[i] instanceof nexacro.Edit)	{
 54 				this._gfnSetEditMsClear(arrComp[i]);
 55 			}
 56 			
 57 			if (arrComp[i] instanceof nexacro.Calendar)
 58 			{
 59 				// 월달력 Popup Div 호출 이벤트 추가
 60 				if (arrComp[i].uCalType == "MM") {
 61 					arrComp[i].addEventHandler("ondropdown", this.gfnCalMMOndropdown, this);
 62 				}
 63 			}
 64 		}
 65 	}
 66    
 67 	// 다국어 처리
 68 	this.gfnInitLang(objForm);
 69 	
 70 	// 화면 loading 시간 측정
 71 	if (objForm.parent.name == "divWork")
 72 	{
 73 		var objApp     = objForm.gfnGetApplication();
 74 		var sStartDate = objApp.sStartDate;
 75 		var nStartTime = objApp.nStartTime;
 76 		
 77 		var objDate  = new Date();
 78 		var sEndDate = objDate.getYear()
 79 						+"-"+String(objDate.getMonth()  ).padLeft(2, '0')
 80 						+"-"+String(objDate.getDate()   ).padLeft(2, '0')
 81 						+" "+String(objDate.getHours()  ).padLeft(2, '0')
 82 						+":"+String(objDate.getMinutes()).padLeft(2, '0')
 83 						+":"+String(objDate.getSeconds()).padLeft(2, '0')
 84 						+" "+objDate.getMilliseconds();						
 85 		var nElapseTime = (objDate.getTime() - nStartTime)/1000;
 86 		//trace("gfnFormOnLoad : "+ sStartDate + " - " + nStartTime + " / " + sEndDate + " - " + nElapseTime);		
 87 		
 88 		objForm.parent.parent.staLodingTime.set_text("해당 화면의 loading 시간은 " +  + nElapseTime + " Sec 입니다.");
 89 	}
 90 	
 91 	// 팝업 일때 처리
 92 	if (objForm.opener)
 93 	{
 94 		if (objForm.parent instanceof nexacro.ChildFrame)
 95 		{
 96 			// 키다운 이베트 추가
 97 			objForm.addEventHandler("onkeydown", this.gfnOnkeydown, this);
 98 		}
 99 	}
100 
101 	// QuikView 일때 처리
102 	if (nexacro.getEnvironmentVariable("evQuikView") == "Y") 
103 	{
104 		if (this.gfnIsNull(objForm.opener) && objForm.parent instanceof nexacro.ChildFrame)
105 		{
106 			// 키다운 이베트 추가
107 			objForm.addEventHandler("onkeydown", this.gfnOnkeydown, this);
108 		}
109 	}
110 };

117 	//trace("e.ctrlkey : " + e.ctrlkey + " / e.keycode : " + e.keycode);

129 	}

132 /**
133  * @class left메뉴 클릭시 해당화면 호출함수 <br>
134  * @param {Object} oObj 
135  * @return N/A
136  * @example 
137  */
138 this.gfnCall = function(oObj)
139 {	
140 	if(!this.gfnIsNull(oObj) && typeof(oObj) !=  "object") return;	
141 	
142 	var objApp  = this.gfnGetApplication();
143 	var gdsOpen = objApp.gdsOpenMenu;				//열린 	  dataset	
144 	var ds      = oObj.ds;							//넘어온 dataset
145 	var nRow    = oObj.nRow;						//선택된 현재 row
146 	var oArgs 	= oObj.oArgs;   					//넘어온 arguments
147 	var sMenuId;
148 
149 	if (!this.gfnIsNull( oObj.sMenuId)){
150 		sMenuId = oObj.sMenuId;
151 	}else{
152 		sMenuId = ds.getColumn(nRow, objApp.gvMenuColumns.menuId);
153 	}	
154 	
155 	var winid = gdsOpen.lookup(objApp.gvMenuColumns.menuId, sMenuId, objApp.gvMenuColumns.winId);
156 
157 	if (!this.gfnIsNull(winid))
158 	{
159 		if (objApp.gvMdiFrame.form.isActiveFrame(winid, oArgs) == true)
160 		{
161 			objApp.gvMdiFrame.form.fnMoveTab(winid);
162 			return;
163 		}
164 	}
165 	
166 	//열린메뉴 체크( application.gvMax = 8)	
167 	if( objApp.gvMax <= gdsOpen.getRowCount() ){
168 		          
169 		alert(objApp.gvMax +"개 초과하여 화면을 열수 없습니다");
170 		return false;
171 	}
172 	
173 	this.gfnNewMdi(sMenuId, nRow, oArgs);
174 };

210 /**
211  * @class gdsOpenMenu의 해당 Row의 정보를 기준으로 신규 윈도우 화면을 생성하고 open 시킴 <br>
212  * @param {String} sMenuId - menuId
213  * @param {Number} nRow - gdsOpenMenu의rowpostion
214  * @param {Object} oArgs - arguments
215  * @return N/A
216  */
217 this.gfnNewMdi = function(sMenuId, nRow, oArgs)
218 {	
219 	var objApp   = this.gfnGetApplication();
220 	var gdsOpen  = objApp.gdsOpenMenu;		//열린 dataset
221 	var gdsMenu  = objApp.gdsMenu;
222 	var winid    = "win" + sMenuId + "_" + gdsOpen.getRowCount() + "_" + parseInt(Math.random() * 1000);		
223 	var sPageUrl = gdsMenu.lookupAs(objApp.gvMenuColumns.menuId, sMenuId, objApp.gvMenuColumns.pageUrl);
224 	var sGroupId = gdsMenu.lookupAs(objApp.gvMenuColumns.menuId, sMenuId, objApp.gvMenuColumns.groupId);
225 
226 	// 화면 loading 시간 측정
227 	var objDate = new Date();
228 	var nStartTime = objDate.getTime();
229     var sStartDate = objDate.getYear()
230 						+"-"+String(objDate.getMonth()).padLeft(2, '0')
231 						+"-"+String(objDate.getDate()).padLeft(2, '0')
232 						+" "+String(objDate.getHours()).padLeft(2, '0')
233 						+":"+String(objDate.getMinutes()).padLeft(2, '0')
234 						+":"+String(objDate.getSeconds()).padLeft(2, '0')
235 						+" "+objDate.getMilliseconds();
236 	objApp.nStartTime = nStartTime;
237 	objApp.sStartDate = sStartDate;
238 	
239 	// 다국어 처리
240 	var sColumn  = objApp.gvMenuColumns.menuNm;
241 	var sNowLang = nexacro.getEnvironmentVariable("evLanguage");
242 	if (sNowLang != "KO") {
243 		sColumn = sColumn+sNowLang;
244 	}
245 	
246 	var sMenuNm  = gdsMenu.lookupAs(objApp.gvMenuColumns.menuId, sMenuId, sColumn);
247 	
248 	if(this.gfnIsNull(sPageUrl)) return;		//pageURl 이 없으면 return
249 	this.gfnSetOpenMenuDs(winid, sMenuId, sMenuNm, sPageUrl, sGroupId);	// 열린메뉴 화면 삽입
250 
251 	var objNewWin = new ChildFrame;
252 	objNewWin.init(winid, 0, 0, objApp.gvWorkFrame.getOffsetWidth() - 0, objApp.gvWorkFrame.getOffsetHeight() - 0);
253 	objApp.gvWorkFrame.addChild(winid, objNewWin);
254 
255 	//objNewWin.set_tooltiptext(winid);
256 	objNewWin.arguments = [];
257 	objNewWin.set_dragmovetype("all");
258 	objNewWin.set_showtitlebar(false);
259 	objNewWin.set_resizable(true);
260 	objNewWin.set_openstatus("maximize");
261 	objNewWin.set_titletext(sMenuNm);
262 	objNewWin.set_showcascadetitletext(false);
263 	objNewWin.arguments["winKey"] = winid;
264 	objNewWin.arguments["menuId"] = sMenuId;
265 	objNewWin.arguments["menuNm"] = sMenuNm;
266 	objNewWin.arguments["pageUrl"] = sPageUrl;
267 	objNewWin.arguments["oArgs"] = oArgs;
268 	objNewWin.set_formurl("frame::frameWork.xfdl");
269 
270 	objApp.gvMdiFrame.form.fnAddTab(winid, sMenuNm);   //mdi tab button add	
271 	objApp.gvMdiFrame.form.isActiveFrame(winid);
272 	
273 	objNewWin.show();	
274 };

326 /**
327  * @class 열린화면 데이터셋에 추가 <br>
328  * @param {String} winid
329  * @param {String} menuId
330  * @param {String} strTitle
331  * @param {String} spageUrl
332  * @param {String} sGroupId
333  * @return N/A
334  */
335 this.gfnSetOpenMenuDs = function(winid, menuid, strTitle, spageUrl, sGroupId)
336 {
337 	var objApp  = this.gfnGetApplication();
338 	var gdsOpen = objApp.gdsOpenMenu ;  //열린 dataset
339 	var nRow = gdsOpen.addRow();
340 	gdsOpen.setColumn(nRow, objApp.gvMenuColumns.winId, winid);
341 	gdsOpen.setColumn(nRow, objApp.gvMenuColumns.menuId, menuid);
342 	gdsOpen.setColumn(nRow, objApp.gvMenuColumns.title, strTitle);	
343 	gdsOpen.setColumn(nRow, objApp.gvMenuColumns.groupId, sGroupId);
344 	gdsOpen.setColumn(nRow, "pageUrl", spageUrl);
345 };
